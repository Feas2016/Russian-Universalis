#Conquest of Mamluks
country_event = {

	id = otm.1
	title = "otm.1.title"
	desc = "otm.1.desc"
	picture = AURORA_otm1_eventPicture
	
	trigger = {
		tag = TUR
		is_year = 1500
		government_rank = 3
		religion = sunni
		is_at_war = yes
		any_neighbor_country = {
			tag = MAM
			ai = yes
		}
		any_province = {
			owned_by = MAM
			controlled_by = TUR
		}
		NOT = { is_year = 1550 }
	}
	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = otm.1.a
		ai_chance = { factor = 90 }
		add_treasury = 200
		add_prestige = 10
		add_adm_power = -20
		add_corruption = 0.5
		MAM = {
			every_province = {
				limit = {
					controlled_by = TUR
					owned_by = MAM
				}
				cede_province = TUR
				add_core = TUR
				add_local_autonomy = 10
				add_nationalism = 20				
			}
		}
	}
	
	option = {
		name = otm.1.b
		ai_chance = { factor = 10 }
		add_war_exhaustion = 2
		add_prestige = -10
		add_piety = -0.10
	}
}
