province_event = {
	id = stt_scan.1
	title = "stt_scan.t.1"
	desc = "stt_scan.de.1"
	picture = CITY_DEVELOPMENT_eventPicture
		
	trigger = {
		any_province = {
			OR = {
				province_id = 18
				province_id = 3220
				province_id = 3204
				}
			is_empty = yes
			NOT = { is_colony = yes }
		}
		is_year = 1400
		owner = {
		    OR = {
			    tag = SWE
			    tag = SCA
			    AND = {
				    NOT = { exists = SCA }
				    NOT = { exists = SWE }
				    culture_group = scandinavian
			    }
		    }
		}	
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			is_year = 1500
		}
		modifier = {
			factor = 0.9
			is_year = 1600
		}
		modifier = {
			factor = 1.2
			is_year = 1650
		}
	}
	
	option = {
		name = "stt_scan.a.1"
		random_province = {
			limit = {
				OR = {
					province_id = 18
					province_id = 3220
					province_id = 3204
					}
				is_empty = yes
				}
			create_colony = 1000
		}
	}
}
