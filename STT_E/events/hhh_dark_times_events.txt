	desc = hhh_dark_times_events.0.d

	is_triggered_only = yes
	

	trigger = {
		always = yes			
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = hhh_dark_times_events.0.a
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		add_prestige = 5
		custom_tooltip = hhh_dark_times_events.0.a.tt
		set_country_flag = hhh_dark_ages_f
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = hhh_dark_times_events.1 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
