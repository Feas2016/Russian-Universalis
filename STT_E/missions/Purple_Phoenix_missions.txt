recover_west_anatolia = {
	
	type = country

	category = MIL
	
	target_areas_list = {
		germiyan_area
	}
	
	target_provinces = {
		NOT = { owned_by = ROOT }
		NOT = { owner = { alliance_with = ROOT } }
		NOT = { owner = { is_subject_of = ROOT } }
		NOT = { owner = { overlord_of = ROOT } }
	}
	allow = {
		normal_or_historical_nations = yes
		has_dlc = "Purple Phoenix"
		### STT
		OR = {
			tag = BYZ
			tag = PQR
			tag = ROM
		}
		NOT = { has_country_flag = recovered_west_anatolia }
		is_free_or_tributary_trigger = yes
		owns = 151		# Thrace
		is_at_war = no
		OR = {
			owns = 316 #Kocaeli
			owns = 317 #Bursa
			owns = 318 #Sugla
		}
		calc_true_if = {
			all_province = {
				area = germiyan_area
				NOT = { owned_by = ROOT }
				NOT = { owner = { alliance_with = ROOT } }
				NOT = { owner = { is_subject_of = ROOT } }
				NOT = { owner = { overlord_of = ROOT } }
			}
			amount = 2
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 6
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}
	effect = {
		add_prestige = 10
		add_adm_power = 25
		set_country_flag = recovered_west_anatolia
	}
}

		OR = {
			tag = BYZ

			tag = ROM
		}
		NOT = { has_country_flag = recovered_south_spain }

			owns = 1247

			all_province = {
				OR = {
					province_id = 226

					province_id = 1749
				}
				NOT = { owned_by = ROOT }
				NOT = { owner = { alliance_with = ROOT } }
				NOT = { owner = { is_subject_of = ROOT } }
				NOT = { owner = { overlord_of = ROOT } }
			}
			amount = 3
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2

		}

	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}

		add_prestige = 10
		add_adm_power = 50
		set_country_flag = recovered_south_spain
	}
