recover_greece = {

	category = MIL
	ai_mission = yes
	
	target_provinces_list = {
		144 145 146 147 148
		1773 1853 3003
	}
	
	target_provinces = {
		NOT = { owned_by = ROOT }

		NOT = { owner = { is_subject_of = ROOT } }
		NOT = { owner = { overlord_of = ROOT } }
	}
	
	allow = {
		normal_or_historical_nations = yes
		OR = { tag = BYZ tag = PQR tag = ROM } ### STT
		is_free_or_tributary_trigger = yes
		NOT = { has_country_modifier = byzantium_resurgent }
		NOT = { has_country_flag = retook_greece_byz }
		is_at_war = no
		any_province = {
			OR = {
				province_id = 144
				province_id = 145
				province_id = 146
				province_id = 147
				province_id = 148
				province_id = 1773
				province_id = 1853
				province_id = 3003
			}
			NOT = { owned_by = ROOT }
			NOT = { owner = { alliance_with = ROOT } }
			NOT = { owner = { is_subject_of = ROOT } }
			NOT = { owner = { overlord_of = ROOT } }
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3500
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 6
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}
	effect = {
		set_country_flag = retook_greece_byz
		add_country_modifier = {
			name = "byzantium_resurgent"
			duration = 3650
		}
	}
}
