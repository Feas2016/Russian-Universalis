			factor = 2

		}
		modifier = {
			factor = 2
			STY = { NOT = { has_opinion = { who = HAB value = 0 } } }
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
