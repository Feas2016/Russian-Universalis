#Missions to steer players and AI to bonuses for conquering Holy Sites.

coptic_conquest_of_alexandria = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces_list = {
		358
	}
	
	allow = {
		has_dlc = "Rights of Man"
		religion = coptic
		normal_or_historical_nations = yes
		NOT = { has_global_flag = coptic_alexandria_liberated }
		358 = {
			OR = {
				range = ROOT
				owner = {
					is_neighbor_of = ROOT
				}
			}
			NOT = { owned_by = ROOT }
			owner = {
				NOT = { religion = coptic }
				NOT = { alliance_with = ROOT }
				NOT = { is_subject_of = ROOT }
				NOT = { overlord_of = ROOT }
			}
		}
		is_free_or_tributary_trigger = yes
		OR = {
			ai = no
			num_of_cities = 25
		}
	}
	abort = {
		OR = {
			is_subject_other_than_tributary_trigger = yes
			NOT = { religion = coptic }
			has_global_flag = coptic_alexandria_liberated
			any_target_province = {
				NOT = { owned_by = ROOT }
				owner = {
					religion = coptic
				}
			}
		}
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 0.5
			358 = {
				owner = {
					NOT = { is_neighbor_of = ROOT }
				}
			}
		}
		modifier = {
			factor = 1.5
			mil = 5
		}
		modifier = {
			factor = 2
			OR = {
				owns = 419 #Yerevan
				owns = 1227 #Aksum
				owns = 2313 #Antakya
			}
		}
		modifier = {
			factor = 4
			owns = 1234 #Qasr Ibrim
		}
	}
	effect = {
		tooltip = {
			add_prestige = 10
			gharbia_area = {
				add_province_modifier = {
					name = "the_faithful_liberated"
					duration = 3650
				}
			}
		}
		country_event = { id = coptic_flavor.3 }
	}
}
