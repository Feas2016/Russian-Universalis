# Salted Fish for the Navy
salted_fish_bonus = {
	potential = { 
		OR = {
			salt = 1 
			any_subject_country = { salt = 2 }
			any_known_country = {
				is_colonial_nation_of = ROOT
				salt = 1
			}
		}
		OR = {
			fish = 1 
			any_subject_country = { fish = 2 }
			any_known_country = {
				is_colonial_nation_of = ROOT
				fish = 1
			}
		}
	}

	trigger = {
		OR = {
			salt = 1 
			any_subject_country = { salt = 2 }
			any_known_country = {
				is_colonial_nation_of = ROOT
				salt = 1
			}
		}
		OR = {
			fish = 1 
			any_subject_country = { fish = 2 }
			any_known_country = {
				is_colonial_nation_of = ROOT
				fish = 1
			}
		}
	}

	naval_forcelimit_modifier = 0.05
	naval_maintenance_modifier = -0.025
	naval_attrition = -0.025
	range = 0.10
}

# Shipbuilding Materials
shipbuilding_bonus = {
	potential = { 
		naval_supplies = 1
		num_of_ports = 1
	}

	trigger = { 
		num_of_ports = 1
		grain = 1 
		OR = {
			naval_supplies = 2
			any_subject_country = { naval_supplies = 2 }
			any_known_country = {
				is_colonial_nation_of = ROOT
				naval_supplies = 2
			}
		}		
		OR = {
			iron = 1 
			copper = 1	
		}
		OR = {
			cotton = 1
			cloth = 1
			silk = 1
		}
	}

	global_ship_recruit_speed = -0.025
	global_ship_cost = -0.05
	trade_range_modifier = 0.10
}

# Lots of metals means better firepower
